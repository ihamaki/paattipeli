41      public Kayttoliittyma(int koko) {
42          this.koko = koko;
43          this.peli = new Peli(koko);
44          this.painikkeet1 = new JButton[koko][koko];
45          this.painikkeet2 = new JButton[koko][koko];
46          alustaPainikkeet(painikkeet1);
47          alustaPainikkeet(painikkeet2);
48          pelaaja1 = new Pelikentta(peli, painikkeet1);
49          pelaaja2 = new Pelikentta(peli, painikkeet2);
50          tila = TILA.ALKU;
51      }
52  
53      @Override
54      public void run() {
55          frame = new JFrame("Laivanupotus");
56          frame.setPreferredSize(new Dimension(1000, 500));
57          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
58  
59          if (tila == TILA.PELI) {
60              luoPelinKomponentit(frame.getContentPane());
61              KlikkaustenKuuntelija k = new KlikkaustenKuuntelija(this, peli, painikkeet1,
62                      painikkeet2, pelaaja1, pelaaja2, ohje);
63              lisaaPainikkeilleKuuntelija(k, painikkeet1);
64              lisaaPainikkeilleKuuntelija(k, painikkeet2);
65          } else if (tila == TILA.ALKU) {
66              luoAlkunakymanKomponentit(frame.getContentPane());
67              valitseNaytettavaLauta();
68              LaivojenKuuntelija l = new LaivojenKuuntelija(this, peli, painikkeet1,
69                      painikkeet2, pelaaja1, pelaaja2, napit1, napit2, ohje);
70              lisaaPainikkeilleKuuntelija(l, painikkeet1);
71              lisaaPainikkeilleKuuntelija(l, painikkeet2);
72              lisaaNapeilleKuuntelija(l, napit1);
73              lisaaNapeilleKuuntelija(l, napit2);
74          }
75  
76          frame.pack();
77          frame.setVisible(true);
78          frame.setResizable(false);
79      }

81      /**
82       * Metodi luo käyttöliittymän eri komponentit.
83       *
84       * @param container
85       */
86      private void luoPelinKomponentit(Container container) {
87          container.setLayout(new BorderLayout());
88          JPanel pelaajat = new JPanel(new GridLayout());
89          JPanel kentat = new JPanel(new GridLayout(1, 0));
90  
91          ohje = new JLabel("Pelaajan 1 vuoro", SwingConstants.CENTER);
92          JLabel pel1 = new JLabel("Pelaaja 1", SwingConstants.CENTER);
93          JLabel pel2 = new JLabel("Pelaaja 2", SwingConstants.CENTER);
94  
95          pelaajat.add(pel1);
96          pelaajat.add(pel2);
97          kentat.add(pelaaja1);
98          kentat.add(pelaaja2);
99          pelaaja1.paivita(peli.getLauta1());
100         pelaaja2.paivita(peli.getLauta2());
101 
102         container.add(pelaajat, BorderLayout.NORTH);
103         container.add(kentat, BorderLayout.CENTER);
104         container.add(ohje, BorderLayout.SOUTH);
105     }

107     private void luoAlkunakymanKomponentit(Container container) {
108         container.setLayout(new BorderLayout());
109         JPanel pelaajat = new JPanel(new GridLayout());
110         JPanel kentat = new JPanel(new GridLayout(1, 0));
111         JPanel laivat1 = new JPanel();
112         JPanel laivat2 = new JPanel();
113         laivat1.setLayout(new BoxLayout(laivat1, BoxLayout.Y_AXIS));
114         laivat2.setLayout(new BoxLayout(laivat2, BoxLayout.Y_AXIS));
115 
116         JLabel pel1 = new JLabel("Pelaaja 1", SwingConstants.CENTER);
117         JLabel pel2 = new JLabel("Pelaaja 2", SwingConstants.CENTER);
118         ohje = new JLabel("Pelaaja 1: Aseta laivat laudalle", SwingConstants.CENTER);
119 
120         napit1 = luoNapit(laivat1, true);
121         napit2 = luoNapit(laivat2, false);
122         pelaajat.add(pel1);
123         pelaajat.add(pel2);
124         kentat.add(pelaaja1);
125         kentat.add(pelaaja2);
126         pelaaja2.piilotaKaikki(peli.getLauta2());
127 
128         container.add(pelaajat, BorderLayout.NORTH);
129         container.add(ohje, BorderLayout.SOUTH);
130         container.add(laivat1, BorderLayout.WEST);
131         container.add(laivat2, BorderLayout.EAST);
132         container.add(kentat, BorderLayout.CENTER);
133     }

135     private ArrayList<JButton> luoNapit(Container container, boolean kaytossa) {
136         ArrayList<JButton> napit = new ArrayList<>();
137         napit.add(new JButton("*"));
138         napit.add(new JButton("**"));
139         napit.add(new JButton("***"));
140         napit.add(new JButton("****"));
141         napit.add(new JButton("*****"));
142 
143         for (JButton nappi : napit) {
144             nappi.setEnabled(kaytossa);
145             container.add(nappi);
146         }
147 
148         JButton vaaka = new JButton("vaaka");
149         JButton pysty = new JButton("pysty");
150         vaaka.setEnabled(false);
151         pysty.setEnabled(kaytossa);
152         napit.add(vaaka);
153         napit.add(pysty);
154         container.add(vaaka);
155         container.add(pysty);
156 
157         JButton valmis = new JButton("valmis");
158         valmis.setEnabled(kaytossa);
159         napit.add(valmis);
160         container.add(valmis);
161 
162         return napit;
163     }

173     public void vaihdaTila() {
174         if (tila == TILA.ALKU) {
175             tila = TILA.PELI;
176         } else if (tila == TILA.PELI) {
177             tila = TILA.ALKU;
178         }
179     }

181     public void valitseNaytettavaLauta() {
182         if (peli.getPelattava() == 1) {
183             pelaaja1.naytaLaivat(peli.getLauta1());
184             pelaaja2.piilotaKaikki(peli.getLauta2());
185             for (JButton nappi : napit1) {
186                 nappi.setEnabled(true);
187             }
188             napit1.get(napit1.size() - 3).setEnabled(false);
189             for (JButton nappi : napit2) {
190                 nappi.setEnabled(false);
191             }
192         } else if (peli.getPelattava() == 2) {
193             pelaaja2.naytaLaivat(peli.getLauta2());
194             pelaaja1.piilotaKaikki(peli.getLauta1());
195             for (JButton nappi : napit2) {
196                 nappi.setEnabled(true);
197             }
198             napit2.get(napit2.size() - 3).setEnabled(false);
199             for (JButton nappi : napit1) {
200                 nappi.setEnabled(false);
201             }
202         }
203     }

205     /**
206      * Metodi luo uudet JButtonit kaksiulotteiseen taulukkoon.
207      *
208      * @param painikkeet Kaksiulotteinen taulukko, johon uudet napit lisätään
209      */
210     public void alustaPainikkeet(JButton[][] painikkeet) {
211         for (int i = 0; i < koko; i++) {
212             for (int j = 0; j < koko; j++) {
213                 JButton painike = new JButton();
214                 painikkeet[i][j] = painike;
215             }
216         }
217     }

219     /**
220      * Metodi lisää kuuntelijan jokaiselle JButtonille.
221      *
222      * @param kuuntelija Klikkauksia kuunteleva ja niihin reagoiva kuuntelija
223      * @param painikkeet Kaksiulotteinen JButtonit sisältävä taulukko
224      */
225     public void lisaaPainikkeilleKuuntelija(ActionListener kuuntelija,
226             JButton[][] painikkeet) {
227         for (int i = 0; i < painikkeet.length; i++) {
228             for (int j = 0; j < painikkeet.length; j++) {
229                 painikkeet[i][j].addActionListener(kuuntelija);
230             }
231         }
232     }
233 
234     public void lisaaNapeilleKuuntelija(ActionListener kuuntelija, ArrayList<JButton> napit) {
235         for (JButton nappi : napit) {
236             nappi.addActionListener(kuuntelija);
237         }
238     }
