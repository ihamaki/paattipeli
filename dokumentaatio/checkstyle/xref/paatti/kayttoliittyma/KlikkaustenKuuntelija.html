1   package paatti.kayttoliittyma;
2   
3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import javax.swing.JButton;
6   import javax.swing.JLabel;
7   import paatti.logiikka.Lauta;
8   import paatti.logiikka.Peli;
9   
10  /**
11   * Luokka reagoi pelilaudan graafisen käyttöliittymän painikkeiden klikkauksiin,
12   * ja päivittää pelitilanteen niiden perusteella.
13   */
14  public class KlikkaustenKuuntelija implements ActionListener {
15  
16      private Kayttoliittyma kayttoliittyma;
17      private Peli peli;
18      private Lauta lauta1;
19      private Lauta lauta2;
20      private JButton[][] painikkeet1;
21      private JButton[][] painikkeet2;
22      private Pelikentta pelaaja1;
23      private Pelikentta pelaaja2;
24      private JLabel pelinTila;
25  
26      public KlikkaustenKuuntelija(Kayttoliittyma kayttoliittyma, Peli peli, JButton[][] painikkeet1,
27              JButton[][] painikkeet2, Pelikentta pelaaja1, Pelikentta pelaaja2, JLabel pelinTila) {
28          this.kayttoliittyma = kayttoliittyma;
29          this.peli = peli;
30          this.lauta1 = peli.getLauta1();
31          this.lauta2 = peli.getLauta2();
32          this.painikkeet1 = painikkeet1;
33          this.painikkeet2 = painikkeet2;
34          this.pelaaja1 = pelaaja1;
35          this.pelaaja2 = pelaaja2;
36          this.pelinTila = pelinTila;
37      }
38  
39      @Override
40      public void actionPerformed(ActionEvent e) {
41          if (peli.onkoPeliPaattynyt()) {
42              return;
43          }
44          int pelattava = peli.getPelattava();
45          if (pelattava == 1) {
46              haePelatunRuudunKoordinaatit(lauta1, painikkeet1, pelaaja1, e);
47          } else if (pelattava == 2) {
48              haePelatunRuudunKoordinaatit(lauta2, painikkeet2, pelaaja2, e);
49          }
50      }
51  
52      /**
53       * Metodi hakee klikatun ruudun koordinaatit.
54       *
55       * @param lauta Lauta, johon tällä metodilla haettujen koordinaattien
56       * perusteella ammutaan
57       * @param painikkeet Taulukko JButtoneista, jotka metodi käy läpi
58       * @param pelaaja Pelikenttä, joka päivitetään
59       * @param e Klikkaustapahtuma
60       */
61      public void haePelatunRuudunKoordinaatit(Lauta lauta, JButton[][] painikkeet,
62              Pelikentta pelaaja, ActionEvent e) {
63          for (int i = 0; i < painikkeet.length; i++) {
64              for (int j = 0; j < painikkeet.length; j++) {
65                  if (painikkeet[i][j] == e.getSource()) {
66                      ammu(i, j, lauta, pelaaja);
67                  }
68              }
69          }
70      }
71  
72      /**
73       * Metodi päivittää ammutun ruudun tietoja logiikan Peli-luokkaa kutsumalla.
74       * Samalla metodi päivittää myös pelilaudan graafisen esityksen kutsumalla
75       * Pelikenttä-luokan päivitysmetodia.
76       *
77       * @param x Ammuttavan ruudun x-koordinaatti
78       * @param y Ammuttavan ruudun y-koordinaatti
79       * @param lauta Pelattavana oleva lauta
80       * @param pelaaja Pelikenttä, joka päivitetään
81       */
82      public void ammu(int x, int y, Lauta lauta, Pelikentta pelaaja) {
83          lauta.ammu(x, y);
84          paivitaPelinTila();
85          pelaaja.paivita(lauta);
86      }
87  
88      /**
89       * Päivittää pelivuorossa olevan pelaajan pelitilannetta kuvaavaan
90       * tekstikenttään ja vaihtaa pelattavana olevan laudan. Jos peli on
91       * päättynyt, ilmoittaa voittajan.
92       */
93      public void paivitaPelinTila() {
94          if (peli.onkoPeliPaattynyt()) {
95              peli.vaihdaPelattava();
96              pelinTila.setText("Pelaaja " + peli.getPelattava() + " voitti!");
97          } else {
98              pelinTila.setText("Pelaajan " + peli.getPelattava() + " vuoro");
99              peli.vaihdaPelattava();
100         }
101     }
102
