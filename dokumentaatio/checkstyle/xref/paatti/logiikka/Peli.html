3   /**
4    * Peli-luokka sisältää pelin perustoiminnallisuuden tarjoavat metodit, kuten
5    * tietyn ruudun ampumisen pelilaudalla. Molempien pelaajien pelilautoihin 
6    * päästään käsiksi tämän luokan kautta. Luokka sisältää myös metodit pelitilanteen
7    * tarkistamiseksi, eli onko toinen pelaaja hävinnyt ja onko peli päättynyt.
8    */
9   
10  public class Peli {
11  
12      private Lauta lauta1;
13      private Lauta lauta2;
14      private boolean peliPaattynyt;
15  
16      public Peli(int koko) {
17          this.lauta1 = new Lauta(koko);
18          this.lauta2 = new Lauta(koko);
19          this.peliPaattynyt = false;
20      }
21  
22      public Lauta getLauta1() {
23          return lauta1;
24      }
25  
26      public Lauta getLauta2() {
27          return lauta2;
28      }
29  
30      public void pelaa() {
31          System.out.println("Tervetuloa peliin!");
32      }
33  
34      public void ammu(Lauta lauta, int x, int y) {
35          lauta.ammu(x, y);
36      }
37  
38      /**
39       * Metodi tarkistaa, onko pelaaja hävinnyt tarkistamalla pelaajan laudan 
40       * laivojen tilanteen.
41       * 
42       * @param lauta Tarkistettava pelilauta
43       * @return true jos kaikki laudan laivat on tuhottu, false muuten
44       */
45      public boolean onkoHavinnyt(Lauta lauta) {
46          return lauta.kaikkiLaivatTuhottu();
47      }
48  
49      public boolean onkoPeliPaattynyt() {
50          if (onkoHavinnyt(lauta1) || onkoHavinnyt(lauta2)) {
51              return true;
52          }
53  
54          return false;
55      }
56  
57  }
