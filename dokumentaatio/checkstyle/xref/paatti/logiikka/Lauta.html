1   package paatti.logiikka;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * Lauta pitää sisällään kaksiulotteisen taulukon pelilaudan ruuduista, sekä
8    * listana tiedon laudalle lisätyistä laivoista. Lauta tarjoaa metodit peli-
9    * tilanteen muuttamiseen ja päivittämiseen.
10   */
11  
12  public class Lauta {
13  
14      private int koko;
15      private Ruutu[][] ruudut;
16      private List<Laiva> laivat;
17      
18      /**
19       * Laudan konstruktori. 
20       * 
21       * @param koko pelilaudan yhden sivun pituus
22       */
23      public Lauta(int koko) {
24          this.koko = koko;
25          this.ruudut = new Ruutu[koko][koko];
26          alustaLauta();
27          this.laivat = new ArrayList<>();
28      }
29  
30      public int getKoko() {
31          return koko;
32      }
33  
34      public Ruutu[][] getRuudut() {
35          return ruudut;
36      }
37  
38      public List<Laiva> getLaivat() {
39          return laivat;
40      }
41  
42      public void setLaivat(List<Laiva> laivat) {
43          this.laivat = laivat;
44      }
45      
46      
47      public void ammu(int x, int y) {
48          if (x < 0 || y < 0 || x >= koko || y >= koko || ruudut[x][y].getAmmuttu()) {
49              return;
50          }
51          if (ruudut[x][y].getLaiva() != null) {
52              ruudut[x][y].getLaiva().ammu();
53              ruudut[x][y].setTuhoutunut(true);
54          }
55          ruudut[x][y].setAmmuttu(true);
56      }
57  
58      public void lisaaLaivat() {
59          // laivojen lisäys käyttöliittymän avulla
60          Laiva eka = new Laiva(2);
61          ruudut[1][2].setLaiva(eka);
62          ruudut[1][3].setLaiva(eka);
63          laivat.add(eka);
64      }
65      
66      /**
67       * Metodi tarkistaa pelilaudan laivojen statuksen.
68       * 
69       * @return true jos kaikki pelilaudan laivat tuhottu, muuten false
70       */
71      public boolean kaikkiLaivatTuhottu() {
72          int tuhottu = 0;
73          for (Laiva laiva : laivat) {
74              if (laiva.getTuhoutunut()) {
75                  tuhottu++;
76              }
77          }
78          return tuhottu == laivat.size();
79      }
80  
81      public void alustaLauta() {
82          for (int rivi = 0; rivi < koko; rivi++) {
83              for (int sarake = 0; sarake < koko; sarake++) {
84                  this.ruudut[rivi][sarake] = new Ruutu(rivi, sarake);
85              }
86          }
87      }
88  
89      @Override
90      public boolean equals(Object o) {
91          if (o == null) {
92              return false;
93          }
94          if (this.getClass() != o.getClass()) {
95              return false;
96          }
97  
98          Lauta verrattava = (Lauta) o;
99  
100         for (int i = 0; i < laivat.size(); i++) {
101             if (this.laivat.get(i) != verrattava.laivat.get(i)) {
102                 System.out.println("laiva");
103                 return false;
104             }
105         }
106 
107         return this.koko == verrattava.koko;
108     }
109 }
